269 template <typename Container>
270 void splitString(Container& _container, const std::string& _in,
271  const char* const _delimiters = " \t\n")
272 {
273  const std::string::size_type len = _in.length();
274  std::string::size_type i = 0;
275 
276  while ( i < len )
277  {
278  // eat leading whitespace
279  i = _in.find_first_not_of (_delimiters, i);
280  if (i == std::string::npos)
281  return; // nothing left but white space
282 
283  // find the end of the token
284  std::string::size_type j = _in.find_first_of (_delimiters, i);
285 
286  // push token
287  if (j == std::string::npos)
288  {
289  _container.push_back (_in.substr(i));
290  return;
291  }
292  else
293  _container.push_back (_in.substr(i, j-i));
294 
295  // set up for next loop
296  i = j + 1;
297  }
298 }
